#!/bin/bash

if [ "$#" -ne 1 ]; then
    echo "Usage: $0 <file.txt>"
    exit 1
fi

input_file="$1"

if [ ! -e "$input_file" ]; then
    echo "Input file does not exist."
    exit 1
fi

output_file="${input_file%.txt}_uppercase.txt"

tr '[:lower:]' '[:upper:]' < "$input_file" > "$output_file"
echo "Conversion complete. Result saved to $output_file"
